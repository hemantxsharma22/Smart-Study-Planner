Feature Breakdown

8 Functional Modules

Each module handles one clearly defined domain. No module is mandatory — the design accommodates incremental, real-world usage patterns.

1

Student Management

Registration, profile management, goal setting. Student ID is the O(1) dictionary key for all downstream operations.

2

Subject Management

Add subjects (priority 1–5), add topics, search recursively by name, sort by priority with lambda.

3

Study Schedule

Daily / weekly / monthly plans. @log_action decorates generate_plan(). Sort plans by study hours.

4

Assignment Management

Add assignments with deadlines, detect overdue items via datetime, mark completed, log every action.

5

Goal Management

Store short-term and long-term goals as text in the student's profile. View and add at any point.

6

Reminder Management

Type-safe reminders (Study / Exam / Assignment / Goal) validated against TYPES tuple. Datetime comparison triggers alerts.

7

Performance Analysis

Productivity score = (tasks / hours) × 10. Grade A+ to D. Class method for aggregate statistics.

8

Report Generation

Daily / Weekly / Monthly / Performance reports. @log_report decorator. Auto-named CSV export.

from abc import ABC, abstractmethod

class Person(ABC):
    """Abstract base class representing any person in the system."""

    def __init__(self, person_id, name, email, mobile_number):
        self.__person_id     = person_id   # name-mangled: _Person__person_id
        self.__name          = name
        self.__email         = email
        self.__mobile_number = mobile_number

    @property
    def person_id(self): return self.__person_id

    @property
    def name(self): return self.__name

    @property
    def email(self): return self.__email

    @abstractmethod           # subclasses MUST implement this
    def display_details(self): pass

    def update_details(self, name=None, email=None, mobile=None):
        if name:   self.__name          = name
        if email:  self.__email         = email
        if mobile: self.__mobile_number = mobile

    def __str__(self):
        return f"Person[{self.__person_id}]: {self.__name} <{self.__email}>"

    def __repr__(self):
        return f"Person(id={self.__person_id!r}, name={self.__name!r})"
from person import Person

class Student(Person):                   # Inheritance from abstract Person
    """Represents a student; inherits from Person."""

    def __init__(self, person_id, name, email, mobile_number,
                 student_id, course, semester):
        super().__init__(person_id, name, email, mobile_number)
        self.student_id       = student_id
        self.course           = course
        self.semester         = semester
        self.study_goals      = []          # list — ordered, allows dupes
        self.schedule         = []          # list of StudyPlan objects
        self.completed_topics = set()       # set — auto-deduplication O(1) lookup

    def display_details(self):             # Polymorphism — own implementation
        print(f"Student: {self.name}  │  Course: {self.course}  │  Sem: {self.semester}")

    def create_schedule(self, study_plan):
        self.schedule.append(study_plan)
        print(f"Schedule created for {self.name}.")

    def track_progress(self):              # Returns dict — separates data from display
        completed = len(self.completed_topics)
        return {"completed_topics": completed, "goals": self.study_goals}

    def __str__(self):
        return f"Student[{self.student_id}]: {self.name} - {self.course}, Sem {self.semester}"
import json, os
from datetime import datetime, date

class SmartStudyPlanner:                 # Composition — no inheritance
    """Core application class: aggregates all domain objects."""

    def __init__(self):
        self.students            = {}      # dict: student_id → Student (O(1) lookup)
        self.subjects            = []
        self.study_plans         = []
        self.assignments         = []
        self.reminders           = []
        self.reports             = []
        self._subject_categories = set()  # _ prefix = internal use convention

    # ── Lambda-based sorts ──
    def sorted_subjects(self):
        return sorted(self.subjects,    key=lambda s: s.priority)

    def sorted_assignments(self):
        return sorted(self.assignments,  key=lambda a: a.due_date)

    def sorted_plans(self):
        return sorted(self.study_plans,  key=lambda p: p.study_hours)

    # ── Generator — memory-efficient iteration ──
    def pending_assignments_gen(self):
        return (a for a in self.assignments if a.status == 'Pending')

    # ── List comprehension ──
    def pending_assignments_list(self):
        return [a.title for a in self.assignments if a.status == 'Pending']

    # ── Recursive search ──
    def search_subject(self, name, subjects=None, index=0):
        if subjects is None: subjects = self.subjects
        if index >= len(subjects): return None   # base case 1
        if subjects[index].subject_name.lower() == name.lower():
            return subjects[index]                  # base case 2
        return self.search_subject(name, subjects, index + 1)  # recurse

    # ── JSON persistence ──
    def save_data(self, filepath='data.json'):
        data = {
            'students':    [{'id': s.student_id, 'name': s.name, ...} for s in self.students.values()],
            'assignments': [{'id': a.assignment_id, 'title': a.title, ...} for a in self.assignments]
        }
        with open(filepath, 'w') as f: json.dump(data, f, indent=2)

    def __len__(self): return len(self.subjects)
class Performance:
    """Tracks and analyses student academic performance."""

    def __init__(self, performance_id, study_hours, completed_tasks):
        self.performance_id  = performance_id
        self.study_hours     = study_hours
        self.completed_tasks = completed_tasks

    @staticmethod                          # No instance needed — pure calculation
    def productivity_score(study_hours, completed_tasks):
        if study_hours == 0: return 0       # ZeroDivisionError guard
        return round((completed_tasks / study_hours) * 10, 2)

    @staticmethod
    def calculate_grade(score):
        if   score >= 9: return 'A+'
        elif score >= 8: return 'A'
        elif score >= 7: return 'B'
        elif score >= 6: return 'C'
        else:            return 'D'

    @classmethod                           # Receives cls — aggregates instances
    def overall_statistics(cls, performances):
        total_hours = sum(p.study_hours     for p in performances)
        total_tasks = sum(p.completed_tasks for p in performances)
        return {'total_hours': total_hours, 'total_tasks': total_tasks}

    def generate_analysis(self):
        score = self.calculate_productivity()
        grade = self.calculate_grade(score)
        print(f"Performance[{self.performance_id}]: Score = {score},  Grade = {grade}")
from datetime import date

def log_submission(func):           # Decorator factory
    """Logs assignment-related actions without touching business logic."""
    def wrapper(*args, **kwargs):
        print(f"[LOG] Assignment action: {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

class Assignment:
    """Manages lifecycle of academic work from creation to completion."""

    def __init__(self, assignment_id, title, due_date, status='Pending'):
        self.assignment_id = assignment_id
        self.title         = title
        self.due_date      = due_date
        self.status        = status

    @log_submission                    # Decorator applied — logs every add
    def add_assignment(self):
        print(f"Assignment '{self.title}' added. Deadline: {self.due_date}")

    @log_submission                    # Logs every completion too
    def mark_completed(self):
        self.status = 'Completed'
        print(f"Assignment '{self.title}' marked as completed.")

    def is_overdue(self):               # Only Pending items can be overdue
        return date.today() > self.due_date and self.status != 'Completed'
